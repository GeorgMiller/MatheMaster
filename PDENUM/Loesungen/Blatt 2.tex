\end{align*}

Außerdem ist das Minimum $u\in W$ von \eqref{eq1} eindeutig.

\begin{align*}

\end{align*}

\begin{align}\label{Hinweis2.5}
2\cdot a\cdot b\leq\gamma\cdot a^2+\frac{b^2}{\gamma}\qquad\forall a,b\in\R,\forall\gamma>0
\end{align}
als gegeben.


\end{proof}
